package com.fiscalapi;

import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;
import java.net.URI;
import com.google.gson.Gson;
import java.time.LocalDate;
import java.util.HashMap;
import java.util.Map;

public class FiscalApiClient {
    private final String apiKey;
    private final String baseUrl = "https://api.fiscal.service/v1";
    private final HttpClient httpClient;
    private final Gson gson;

    public FiscalApiClient(String apiKey) {
        this.apiKey = apiKey;
        this.httpClient = HttpClient.newHttpClient();
        this.gson = new Gson();
    }

    public String crearFactura() throws Exception {
        try {
            Map<String, Object> factura = new HashMap<>();
            factura.put("tipo_comprobante", "A");
            factura.put("punto_venta", 1);
            factura.put("concepto", 1);
            factura.put("doc_tipo", 80);
            factura.put("doc_nro", "30712345678");
            factura.put("fecha", LocalDate.now().toString());
            factura.put("imp_total", 1000.00);
            factura.put("imp_neto", 826.45);
            factura.put("imp_iva", 173.55);

            String jsonBody = gson.toJson(factura);

            HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create(baseUrl + "/comprobantes"))
                .header("Content-Type", "application/json")
                .header("Authorization", "Bearer " + apiKey)
                .POST(HttpRequest.BodyPublishers.ofString(jsonBody))
                .build();

            HttpResponse<String> response = httpClient.send(request, 
                HttpResponse.BodyHandlers.ofString());

            if (response.statusCode() == 200) {
                return response.body();
            } else {
                throw new RuntimeException("Error: " + response.body());
            }
        } catch (Exception e) {
            throw new RuntimeException("Error al crear factura: " + e.getMessage());
        }
    }

    // Ejemplo de uso
    public static void main(String[] args) {
        FiscalApiClient client = new FiscalApiClient("TU_API_KEY");
        try {
            String resultado = client.crearFactura();
            System.out.println("Factura creada: " + resultado);
        } catch (Exception e) {
            System.err.println("Error: " + e.getMessage());
        }
    }
}

CREATE CLASS FiscalApiClient
    PROTECTED:
        VAR cApiKey
        VAR cBaseUrl
        VAR oCurl

    EXPORTED:
        METHOD New(cKey)
        METHOD CrearFactura()
        METHOD Destroy()
ENDCLASS

METHOD New(cKey) CLASS FiscalApiClient
    ::cApiKey := cKey
    ::cBaseUrl := "https://api.fiscal.service/v1"
    ::oCurl := curl_easy_init()
    RETURN Self

METHOD Destroy() CLASS FiscalApiClient
    curl_easy_cleanup(::oCurl)
    RETURN NIL

METHOD CrearFactura() CLASS FiscalApiClient
    LOCAL hFactura := HHash()
    LOCAL cJson, cResponse, nStatus
    LOCAL aHeaders := {}

    TRY
        // Preparar datos de la factura
        hFactura["tipo_comprobante"] := "A"
        hFactura["punto_venta"] := 1
        hFactura["concepto"] := 1
        hFactura["doc_tipo"] := 80
        hFactura["doc_nro"] := "30712345678"
        hFactura["fecha"] := DToC(Date())
        hFactura["imp_total"] := 1000.00
        hFactura["imp_neto"] := 826.45
        hFactura["imp_iva"] := 173.55

        // Convertir a JSON
        cJson := hb_jsonEncode(hFactura)

        // Configurar headers
        AAdd(aHeaders, "Content-Type: application/json")
        AAdd(aHeaders, "Authorization: Bearer " + ::cApiKey)

        // Configurar CURL
        curl_easy_setopt(::oCurl, HB_CURLOPT_URL, ::cBaseUrl + "/comprobantes")
        curl_easy_setopt(::oCurl, HB_CURLOPT_POST, 1)
        curl_easy_setopt(::oCurl, HB_CURLOPT_POSTFIELDS, cJson)
        curl_easy_setopt(::oCurl, HB_CURLOPT_HTTPHEADER, aHeaders)

        // Ejecutar request
        cResponse := curl_easy_perform(::oCurl)
        nStatus := curl_easy_getinfo(::oCurl, HB_CURLINFO_RESPONSE_CODE)

        IF nStatus == 200
            RETURN cResponse
        ELSE
            RETURN "Error: " + hb_ntos(nStatus) + " - " + cResponse
        ENDIF

    CATCH oError
        RETURN "Error: " + oError:description
    END

public function string crear_factura();
    string ls_json, ls_response, ls_error
    JSONPackage lnv_json
    JSONObject lnv_factura
    
    try
        // Crear objeto JSON para la factura
        lnv_json = create JSONPackage
        lnv_factura = lnv_json.CreateJSONObject()
        
        // Configurar datos de la factura
        lnv_factura.AddString("tipo_comprobante", "A")
        lnv_factura.AddNumber("punto_venta", 1)
        lnv_factura.AddNumber("concepto", 1)
        lnv_factura.AddNumber("doc_tipo", 80)
        lnv_factura.AddString("doc_nro", "30712345678")
        lnv_factura.AddString("fecha", String(Today(), "yyyy-mm-dd"))
        lnv_factura.AddNumber("imp_total", 1000.00)
        lnv_factura.AddNumber("imp_neto", 826.45)
        lnv_factura.AddNumber("imp_iva", 173.55)
        
        ls_json = lnv_factura.GetJsonString()
        
        // Configurar cliente HTTP
        io_http_client = create OLEObject
        io_http_client.ConnectToNewObject("MSXML2.XMLHTTP")
        
        // Configurar y enviar request
        io_http_client.Open("POST", is_base_url + "/comprobantes", false)
        io_http_client.SetRequestHeader("Content-Type", "application/json")
        io_http_client.SetRequestHeader("Authorization", "Bearer " + is_api_key)
        io_http_client.Send(ls_json)
        
        if io_http_client.Status = 200 then
            ls_response = io_http_client.ResponseText
        else
            ls_error = "Error " + String(io_http_client.Status) + ": " + io_http_client.ResponseText
            throw ls_error
        end if
        
        return ls_response
        
    catch (RuntimeError rte)
        return "Error: " + rte.GetMessage()
        
    finally
        if IsValid(lnv_json) then destroy lnv_json
        if IsValid(io_http_client) then destroy io_http_client
    end try
end function

Private Sub GuardarFacturaEnDB(jsonResponse As String)
    Dim db As DAO.Database
    Dim rs As DAO.Recordset
    
    Set db = CurrentDb()
    Set rs = db.OpenRecordset("Facturas")
    
    With rs
        .AddNew
        !fecha = Date
        !respuesta_api = jsonResponse
        .Update
    End With
    
    rs.Close
    Set rs = Nothing
    Set db = Nothing
End Sub

Public Function CrearFactura(ws As Worksheet) As String
    On Error GoTo ErrorHandler
    
    Dim xmlHttp As Object
    Dim jsonBody As String
    Dim response As String
    Dim lastRow As Long
    
    ' Inicializar configuración

    m_Config.BaseUrl = "https://api.fiscal.service/v1"
    
    ' Obtener última fila con datos
    lastRow = ws.Cells(ws.Rows.Count, "A").End(xlUp).Row
    
    ' Crear objeto para request HTTP
    Set xmlHttp = CreateObject("MSXML2.XMLHTTP")
    
    ' Preparar JSON de la factura
    jsonBody = "{" & _
        """tipo_comprobante"": ""A"", " & _
        """punto_venta"": " & ws.Range("B2").Value & ", " & _
        """concepto"": " & ws.Range("C2").Value & ", " & _
        """doc_tipo"": " & ws.Range("D2").Value & ", " & _
        """doc_nro"": """ & ws.Range("E2").Value & """, " & _
        """fecha"": """ & Format(Date, "yyyy-mm-dd") & """, " & _
        """imp_total"": " & ws.Range("F2").Value & ", " & _
        """imp_neto"": " & ws.Range("G2").Value & ", " & _
        """imp_iva"": " & ws.Range("H2").Value & "}"
    
    ' Enviar request
    xmlHttp.Open "POST", m_Config.BaseUrl & "/comprobantes", False
    xmlHttp.setRequestHeader "Content-Type", "application/json"
    xmlHttp.setRequestHeader "Authorization", "Bearer " & m_Config.ApiKey
    xmlHttp.send jsonBody
    
    ' Procesar respuesta
    If xmlHttp.Status = 200 Then
        CrearFactura = xmlHttp.responseText
        
        ' Guardar respuesta en la hoja
        ws.Range("I" & lastRow + 1).Value = "CAE: " & ParseJSON(xmlHttp.responseText)
    Else
        Err.Raise xmlHttp.Status, "CrearFactura", xmlHttp.responseText
    End If
    
    Exit Function

ErrorHandler:
    CrearFactura = "Error: " & Err.Description
End Function

Private Function ParseJSON(jsonString As String) As String
    ' Función simple para extraer el CAE del JSON
    Dim regex As Object
    Set regex = CreateObject("VBScript.RegExp")
    
    regex.Pattern = """cae"":""(\d+)"""
    regex.Global = True
    
    If regex.Test(jsonString) Then
        ParseJSON = regex.Execute(jsonString)(0).SubMatches(0)
    End If
End Function
