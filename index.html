' Ejemplo con Visual Basic 6
Option Explicit

Private Type FiscalConfig
    ApiKey As String
    Environment As String
End Type

Private m_Config As FiscalConfig
Private m_HttpRequest As Object

Private Sub Class_Initialize()
    Set m_HttpRequest = CreateObject("MSXML2.XMLHTTP")
    m_Config.ApiKey = "TU_API_KEY"
    m_Config.Environment = "production"
End Sub

Public Function CrearFactura() As String
    On Error GoTo ErrorHandler
    
    Dim jsonBody As String
    jsonBody = "{" & _
        """tipo_comprobante"": ""A"", " & _
        """punto_venta"": 1, " & _
        """concepto"": 1, " & _
        """doc_tipo"": 80, " & _
        """doc_nro"": ""30712345678"", " & _
        """fecha"": """ & Format(Date, "yyyy-mm-dd") & """, " & _
        """imp_total"": 1000.00, " & _
        """imp_neto"": 826.45, " & _
        """imp_iva"": 173.55" & _
    "}"
    
    m_HttpRequest.Open "POST", "https://api.fiscal.service/v1/comprobantes", False
    m_HttpRequest.setRequestHeader "Authorization", "Bearer " & m_Config.ApiKey
    m_HttpRequest.setRequestHeader "Content-Type", "application/json"
    m_HttpRequest.send jsonBody
    
    If m_HttpRequest.Status = 200 Then
        CrearFactura = m_HttpRequest.responseText
    Else
        Err.Raise m_HttpRequest.Status, "CrearFactura", m_HttpRequest.responseText
    End If
    
    Exit Function
    
ErrorHandler:
    CrearFactura = "Error: " & Err.Description
End Function

// Ejemplo con C#
using System.Net.Http;
using System.Text;
using System.Text.Json;
using System.Threading.Tasks;

namespace FiscalApi
{
    public class FiscalApiClient
    {
        private readonly HttpClient _httpClient;
        private readonly String _baseUrl = "https://api.fiscal.service/v1";

        public FiscalApiClient(String apiKey)
        {
            _httpClient = new HttpClient();
            _httpClient.DefaultRequestHeaders.Add("Authorization", $"Bearer {apiKey}");
        }

        public async Task<String> CrearFactura()
        {
            try
            {
                var factura = new
                {
                    tipo_comprobante = "A",
                    punto_venta = 1,
                    concepto = 1,
                    doc_tipo = 80,
                    doc_nro = "30712345678",
                    fecha = DateTime.Now.ToString("yyyy-MM-dd"),
                    imp_total = 1000.00,
                    imp_neto = 826.45,
                    imp_iva = 173.55
                };

                var json = JsonSerializer.Serialize(factura);
                var content = new StringContent(json, Encoding.UTF8, "application/json");

                var response = await _httpClient.PostAsync($"{_baseUrl}/comprobantes", content);
                response.EnsureSuccessStatusCode();

                return await response.Content.ReadAsStringAsync();
            }
            catch (Exception ex)
            {
                return $"Error: {ex.Message}";
            }
        }
    }
}

// Uso del cliente

var resultado = await client.CrearFactura();
// Ejemplo con Delphi
unit FiscalApi;

interface

uses
  System.SysUtils, System.Classes, System.JSON,
  System.Net.HttpClient, System.Net.URLClient;

type
  TFiscalApiClient = class
  private
    FHttpClient: THTTPClient;
    FApiKey: string;
    FBaseUrl: string;
  public
    constructor Create(const ApiKey: string);
    destructor Destroy; override;
    function CrearFactura: string;
  end;

implementation

constructor TFiscalApiClient.Create(const ApiKey: string);
begin
  inherited Create;
  FApiKey := ApiKey;
  FBaseUrl := 'https://api.fiscal.service/v1';
  FHttpClient := THTTPClient.Create;
  FHttpClient.CustomHeaders['Authorization'] := 'Bearer ' + ApiKey;
end;

destructor TFiscalApiClient.Destroy;
begin
  FHttpClient.Free;
  inherited;
end;

function TFiscalApiClient.CrearFactura: string;
var
  JsonObj: TJSONObject;
  Response: IHTTPResponse;
begin
  try
    JsonObj := TJSONObject.Create;
    try
      JsonObj.AddPair('tipo_comprobante', 'A');
      JsonObj.AddPair('punto_venta', TJSONNumber.Create(1));
      JsonObj.AddPair('concepto', TJSONNumber.Create(1));
      JsonObj.AddPair('doc_tipo', TJSONNumber.Create(80));
      JsonObj.AddPair('doc_nro', '30712345678');
      JsonObj.AddPair('fecha', '2024-02-21');
      JsonObj.AddPair('imp_total', TJSONNumber.Create(1000.00));
      JsonObj.AddPair('imp_neto', TJSONNumber.Create(826.45));
      JsonObj.AddPair('imp_iva', TJSONNumber.Create(173.55));

      Response := FHttpClient.Post(
        FBaseUrl + '/comprobantes',
        TStringStream.Create(JsonObj.ToString),
        nil
      );

      Result := Response.ContentAsString;
    finally
      JsonObj.Free;
    end;
  except
    on E: Exception do
      Result := 'Error: ' + E.Message;
  end;
end;

end.
